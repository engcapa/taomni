//! Transport spike: does Claude Code's `.mcp.json` `type:"http"` connect to an
//! rmcp Streamable-HTTP server (Bearer-gated, 127.0.0.1)? Exits 0 iff CC
//! actually invokes our `ping` tool end-to-end through the real CLI.

use std::sync::atomic::{AtomicUsize, Ordering};
use std::sync::Arc;

use axum::body::Body;
use axum::extract::Request;
use axum::http::StatusCode;
use axum::middleware::Next;
use axum::response::Response;
use rmcp::handler::server::tool::ToolRouter;
use rmcp::handler::server::wrapper::Json;
use rmcp::model::{ServerCapabilities, ServerInfo};
use rmcp::transport::streamable_http_server::session::local::LocalSessionManager;
use rmcp::transport::streamable_http_server::StreamableHttpService;
use rmcp::{tool, tool_handler, tool_router, ServerHandler};
use serde::Serialize;

static PING_HITS: AtomicUsize = AtomicUsize::new(0);

#[derive(Clone)]
struct Spike {
    tool_router: ToolRouter<Self>,
}

#[derive(Serialize, schemars::JsonSchema)]
struct PingOut {
    pong: String,
}

#[tool_router]
impl Spike {
    fn new() -> Self {
        Self {
            tool_router: Self::tool_router(),
        }
    }

    #[tool(name = "ping", description = "Returns a fixed nonce so the caller can prove it reached this server.")]
    async fn ping(&self) -> Json<PingOut> {
        PING_HITS.fetch_add(1, Ordering::SeqCst);
        Json(PingOut {
            pong: "spike-nonce-7Q3".into(),
        })
    }
}

#[tool_handler]
impl ServerHandler for Spike {
    fn get_info(&self) -> ServerInfo {
        let mut info = ServerInfo::default();
        info.capabilities = ServerCapabilities::builder().enable_tools().build();
        info.instructions = Some("Spike server. Call `ping`.".into());
        info
    }
}

const TOKEN: &str = "spike-bearer-token-abc123";

async fn auth(req: Request, next: Next) -> Result<Response, StatusCode> {
    let ok = req
        .headers()
        .get("authorization")
        .and_then(|v| v.to_str().ok())
        .map(|v| v == format!("Bearer {TOKEN}"))
        .unwrap_or(false);
    if ok {
        Ok(next.run(req).await)
    } else {
        Ok(Response::builder()
            .status(StatusCode::UNAUTHORIZED)
            .body(Body::empty())
            .unwrap())
    }
}

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    let service = StreamableHttpService::new(
        || Ok(Spike::new()),
        Arc::new(LocalSessionManager::default()),
        Default::default(),
    );

    let app = axum::Router::new()
        .nest_service("/mcp", service)
        .layer(axum::middleware::from_fn(auth));

    let listener = tokio::net::TcpListener::bind("127.0.0.1:0").await?;
    let addr = listener.local_addr()?;
    let url = format!("http://{}/mcp", addr);
    println!("SPIKE_SERVER_URL={url}");

    let server = tokio::spawn(async move {
        let _ = axum::serve(listener, app).await;
    });

    // Write a CC-style .mcp.json pointing at us with the Bearer header.
    let mcp_cfg = serde_json::json!({
        "mcpServers": {
            "taomni": {
                "type": "http",
                "url": url,
                "headers": { "Authorization": format!("Bearer {TOKEN}") }
            }
        }
    });
    let cfg_path = std::env::temp_dir().join("rmcp-spike-mcp.json");
    std::fs::write(&cfg_path, serde_json::to_string_pretty(&mcp_cfg)?)?;
    println!("SPIKE_MCP_CONFIG={}", cfg_path.display());

    // Drive the real CC CLI: force it to call our tool.
    let out = tokio::process::Command::new("claude")
        .arg("--model")
        .arg("auto")
        .arg("--mcp-config")
        .arg(&cfg_path)
        .arg("--strict-mcp-config")
        .arg("--allowedTools")
        .arg("mcp__taomni__ping")
        .arg("--print")
        .arg("Call the `ping` tool from the taomni MCP server and reply with the exact value of its `pong` field.")
        .output()
        .await?;

    println!("--- CC STDOUT ---\n{}", String::from_utf8_lossy(&out.stdout));
    println!("--- CC STDERR ---\n{}", String::from_utf8_lossy(&out.stderr));

    server.abort();

    let hits = PING_HITS.load(Ordering::SeqCst);
    println!("PING_HITS={hits}");
    if hits > 0 {
        println!("SPIKE_RESULT=PASS");
        Ok(())
    } else {
        println!("SPIKE_RESULT=FAIL");
        std::process::exit(1)
    }
}
